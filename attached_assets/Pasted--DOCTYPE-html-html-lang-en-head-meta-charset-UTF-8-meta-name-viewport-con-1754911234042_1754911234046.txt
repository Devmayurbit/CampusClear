                    this.insertBefore(spinner, this.firstChild);
                    this.disabled = true;
                    
                    setTimeout(() => {
                        try { this.removeChild(spinner); } catch(e) {}
                        this.innerHTML = '<i class="fas fa-sync-alt mr-2"></i> Refresh Status';
                        this.disabled = false;
                        
                        // Show notification
                        showToast('Status updated successfully', 'success');
                    }, 1500);
                });
            }
            
            // Toast notification function
            function showToast(message, type) {
                const toast = document.createElement('div');
                toast.className = `fixed bottom-4 right-4 px-6 py-3 rounded-md shadow-lg text-white ${
                    type === 'success' ? 'bg-green-600' : 
                    type === 'error' ? 'bg-red-600' : 'bg-blue-600'
                }`;
                toast.textContent = message;
                document.body.appendChild(toast);
                
                setTimeout(() => {
                    toast.style.opacity = '0';
                    setTimeout(() => {
                        try { document.body.removeChild(toast); } catch(e){}
                    }, 300);
                }, 3000);
            }
            
            // Demo functionality for file uploads
            const fileUploadButtons = document.querySelectorAll('button.button-hover');
            fileUploadButtons.forEach(button => {
                // Only attach to buttons that look like upload buttons
                if (button.innerText && button.innerText.trim().toLowerCase().includes('upload')) {
                    button.addEventListener('click', function() {
                        const input = document.createElement('input');
                        input.type = 'file';
                        input.style.display = 'none';
                        input.addEventListener('change', function() {
                            if (this.files && this.files[0]) {
                                showToast(`${this.files[0].name} uploaded successfully`, 'success');
                                const listItem = this.closest('li');
                                if (listItem) {
                                    listItem.classList.remove('bg-yellow-50');
                                    listItem.classList.add('bg-blue-50');
                                }
                            }
                        });
                        document.body.appendChild(input);
                        input.click();
                        document.body.removeChild(input);
                    });
                }
            });
            
            // Department card click events
            const departmentCards = document.querySelectorAll('.department-card');
            departmentCards.forEach(card => {
                card.addEventListener('click', function() {
                    const statusBadge = this.querySelector('.status-badge');
                    const status = statusBadge ? statusBadge.textContent.trim() : '';
                    let message = '';
                    
                    if (status.includes('Cleared')) {
                        message = 'This department has cleared all requirements.';
                    } else if (status.includes('Pending')) {
                        message = 'Some requirements are pending. Click for details.';
                    } else {
                        message = 'There are unresolved issues with this department.';
                    }
                    
                    showToast(message, status.includes('Cleared') ? 'success' : status.includes('Pending') ? 'info' : 'error');
                });
            });
            
            // Add animation classes for cards (staggered)
            const cards = document.querySelectorAll('.bg-white.card-hover');
            cards.forEach((card, index) => {
                card.classList.add('fade-in');
                card.style.animationDelay = `${index * 0.05}s`;
            });
        });
    </script>
</body>
